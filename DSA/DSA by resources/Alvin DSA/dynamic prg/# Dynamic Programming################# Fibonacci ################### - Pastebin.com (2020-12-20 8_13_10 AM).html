  74.     table = [[0] * (n + 1) for _ in range(m + 1)]
  75.     table[1][1] = 1
  76.     for i in range(1, m):
  77.         for j in range(1, n):
  78.             table[i][j + 1] += table[i][j]
  79.             table[i + 1][j] += table[i][j]
  80.     # right-most column
  81.     for i in range(1, m):
  82.         table[i + 1][-1] += table[i][-1]
  83.     # bottom row
  84.     for j in range(1, n):
  85.         table[-1][j + 1] += table[-1][j]
  86.     return table[m][n]
